# 用递归算法实现，数组长度为5且元素的随机数在2-32间不重复的值

function buildArray(arr, length, min, max) {
  let num = Math.floor(Math.random() * (max-min+1))+min
  if (!arr.includes(num)) { arr.push(num) }
  return arr.length === length ? arr : buildArray(arr, length, min, max)
}
let result = buildArray([], 5, 2, 32)
console.log(result)

function delLast(str, target) {
  return str.split('').reverse().join('').replace(target, '').split('').reverse().join('');
}

  return str.replace(reg,'')

# 写一个方法把下划线命名成大驼峰命名法

// 正则
const toCamel = str => {
  str = str.replace(/(\w)/, (match, $1) => `${$1.toUpperCase()}`)
  while(str.match(/\w_\w/)) {
    str = str.replace(/(\w)(_)(\w)/, (match, $1, $2, $3) => `${$1}${$3.toUpperCase()}`)
  }
  return str
}
console.log(toCamel('a_c_def')) // ACDef

# 写一个方法把字符串大小写切换

// 字符串方法
const caseConvert = str => {
  return str.split('').map(s => {
    const code = s.charCodeAt();
    if (code < 65 || code > 122 || code < 90 && code < 97) return s;
    if (code <= 90) {
      return String.formCharCode(code + 32)
    } else {
      return String.formCharCode(code - 32)
    }
  }).join('')
}

const caseConvert = str => {
  return str.split('').map(s => {
    if (s.charCodeAt() <= 90) {
      return s.toLowerCase();
    }
    return s.toUpperCase();
  }).join('')
}

// 凯撒密码
const encodeCaesar = ({str = "", padding = 3}) =>
  !str
    ? str
    : str
        .split("")
        .map((s) => String.fromCharCode(s.charCodeAt() + padding))
        .join("");

const decodeCaesar = ({str = "", padding = 3}) =>
  !str
    ? str
    : str
        .split("")
        .map((s) => String.fromCharCode(s.charCodeAt() - padding))
        .join("");

console.log(encodeCaesar({str: "hello world"}));
console.log(decodeCaesar({str: "khoor#zruog"}));
